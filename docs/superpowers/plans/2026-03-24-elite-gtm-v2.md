# Elite GTM Automation System V2 — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a fully autonomous, multi-channel GTM system that maps the total addressable market, tracks buying signals across 10+ sources, reaches prospects via email + LinkedIn + voice, and closes the loop by routing inbound replies to auto-booked meetings — zero human involvement required.

**Architecture:** A `companies` table becomes the TAM foundation every other system references. Signals (existing + new intent/tech/job-change types) aggregate into per-company heat scores that drive a priority queue. The multi-channel orchestrator fires email → LinkedIn → voice in a 7-day sequence, and inbound replies are classified by Claude and auto-routed to drafted responses or calendar bookings.

**Tech Stack:** Next.js 14 (App Router), Supabase Postgres, Apify (scraping), HeyReach API (LinkedIn automation), Vapi (AI voice), Claude Sonnet 4.6 (classification + drafting), Perplexity sonar-pro (research), Instantly AI (email sequences)

**Dependency Order:** Phase 1 (Foundation) → Phase 2 (TAM + Discovery) → Phase 3 (Signal Expansion) → Phase 4 (Reach Layer) → Phase 5 (Close Layer) → Phase 6 (Orchestration)

// src/lib/tam/company-discovery.ts
// Scrapes company directories to populate the TAM companies table.
// Sources: Crunchbase (via Apify), G2 company listings, LinkedIn company search.

import { runActor, getRunStatus, getDatasetItems } from '@/lib/apify';
import { supabase } from '@/lib/supabase/client';
import { scoreCompany } from './company-scorer';

interface DiscoveredCompany {
  domain: string;
  name: string;
  industry?: string;
  headcount_range?: string;
  funding_stage?: string;
  location?: string;
  source: string;
}

// Apify Crunchbase scraper — finds companies matching ICP criteria
export async function discoverFromCrunchbase(
  keywords: string[],
  limit = 100,
): Promise<DiscoveredCompany[]> {
  const { runId, datasetId } = await runActor('curious_coder/crunchbase-companies-scraper', {
    queries: keywords,
    maxResults: limit,
    includePublicCompanies: false,
  });

  // Poll up to 3 minutes
  for (let i = 0; i < 36; i++) {
    await new Promise((r) => setTimeout(r, 5000));
    const status = await getRunStatus(runId);
    if (status.status === 'SUCCEEDED') {
      const items = await getDatasetItems(datasetId, limit);
      return items.map((item) => ({
        domain: extractDomain(String(item.website || item.homepage_url || '')),
        name: String(item.name || item.organization_name || ''),
        industry: String(item.category_list || item.industry || ''),
        headcount_range: mapHeadcount(String(item.num_employees_enum || '')),
        funding_stage: String(item.last_funding_type || '').toLowerCase().replace(/_/g, '_'),
        location: String(item.country_code || item.headquarters || ''),
        source: 'crunchbase',
      })).filter((c) => c.domain && c.name);
    }
    if (['FAILED', 'TIMED-OUT', 'ABORTED'].includes(status.status)) break;
  }
  return [];
}

// Apify LinkedIn Companies scraper
export async function discoverFromLinkedIn(
  keywords: string[],
  limit = 100,
): Promise<DiscoveredCompany[]> {
  const { runId, datasetId } = await runActor('gbdev/linkedin-company-scraper', {
    searchKeywords: keywords.join(' OR '),
    maxResults: limit,
  });

  for (let i = 0; i < 36; i++) {
    await new Promise((r) => setTimeout(r, 5000));
    const status = await getRunStatus(runId);
    if (status.status === 'SUCCEEDED') {
      const items = await getDatasetItems(datasetId, limit);
      return items.map((item) => ({
        domain: extractDomain(String(item.website || '')),
        name: String(item.name || ''),
        industry: String(item.industry || ''),
        headcount_range: String(item.staffCount || ''),
        location: String(item.headquarterCountry || ''),
        source: 'linkedin',
      })).filter((c) => c.domain && c.name);
    }
    if (['FAILED', 'TIMED-OUT', 'ABORTED'].includes(status.status)) break;
  }
  return [];
}

// Upsert discovered companies into the companies table
export async function upsertCompanies(companies: DiscoveredCompany[]): Promise<number> {
  let inserted = 0;
  for (const c of companies) {
    if (!c.domain || c.domain.length < 4) continue;
    const score = scoreCompany(c);
    const tier = score >= 75 ? 1 : score >= 50 ? 2 : 3;

    const { error } = await supabase.from('companies').upsert({
      domain: c.domain,
      name: c.name || null,
      industry: c.industry || null,
      headcount_range: c.headcount_range || null,
      funding_stage: c.funding_stage || null,
      location: c.location || null,
      tam_score: score,
      tier,
      source: c.source,
      updated_at: new Date().toISOString(),
    }, { onConflict: 'domain', ignoreDuplicates: false });

    if (!error) inserted++;
  }
  return inserted;
}

function extractDomain(url: string): string {
  try {
    const cleaned = url.startsWith('http') ? url : `https://${url}`;
    return new URL(cleaned).hostname.replace(/^www\./, '').toLowerCase();
  } catch {
    return '';
  }
}

function mapHeadcount(raw: string): string {
  if (!raw) return '';
  if (raw.includes('1_10') || raw.includes('1-10')) return '1-10';
  if (raw.includes('11_50') || raw.includes('11-50')) return '11-50';
  if (raw.includes('51_200') || raw.includes('51-200')) return '51-200';
  if (raw.includes('201_500') || raw.includes('201-500')) return '201-500';
  if (raw.includes('501') || raw.includes('1000')) return '500+';
  return raw;
}

// src/lib/tam/company-scorer.ts
// Scores a company 0-100 based on ICP fit criteria.
// Higher = better fit = Tier 1.

interface ScoringInput {
  industry?: string;
  headcount_range?: string;
  funding_stage?: string;
  location?: string;
  tech_stack?: string[];
}

const ICP_INDUSTRIES = ['software', 'saas', 'technology', 'fintech', 'martech', 'sales', 'marketing'];
const ICP_HEADCOUNT = ['11-50', '51-200', '201-500'];
const ICP_FUNDING = ['seed', 'series_a', 'series_b', 'series_c'];
const ICP_LOCATIONS = ['us', 'usa', 'united states', 'united kingdom', 'canada', 'australia'];

export function scoreCompany(company: ScoringInput): number {
  let score = 40; // base

  // Industry fit (+20)
  if (company.industry) {
    const ind = company.industry.toLowerCase();
    if (ICP_INDUSTRIES.some((i) => ind.includes(i))) score += 20;
  }

  // Headcount fit (+15) — sweet spot: 11-500 employees
  if (company.headcount_range && ICP_HEADCOUNT.includes(company.headcount_range)) {
    score += 15;
  }

  // Funding stage fit (+15)
  if (company.funding_stage && ICP_FUNDING.includes(company.funding_stage)) {
    score += 15;
  }

  // Location fit (+10)
  if (company.location) {
    const loc = company.location.toLowerCase();
    if (ICP_LOCATIONS.some((l) => loc.includes(l))) score += 10;
  }

  return Math.min(score, 100);
}

import { NextRequest, NextResponse } from 'next/server';
import { discoverFromCrunchbase, discoverFromLinkedIn, upsertCompanies } from '@/lib/tam/company-discovery';

export const dynamic = 'force-dynamic';
export const maxDuration = 300;

export async function POST(req: NextRequest) {
  const cronSecret = process.env.CRON_SECRET || process.env.WEBHOOK_SECRET;
  if (cronSecret) {
    const auth = req.headers.get('authorization');
    if (auth !== `Bearer ${cronSecret}`) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
  }

  const keywords = (process.env.ICP_DISCOVERY_KEYWORDS || 'B2B SaaS sales software,revenue operations software,CRM software startup').split(',');

  const [crunchbaseCompanies, linkedinCompanies] = await Promise.allSettled([
    discoverFromCrunchbase(keywords, 200),
    discoverFromLinkedIn(keywords, 100),
  ]);

  const all = [
    ...(crunchbaseCompanies.status === 'fulfilled' ? crunchbaseCompanies.value : []),
    ...(linkedinCompanies.status === 'fulfilled' ? linkedinCompanies.value : []),
  ];

  const inserted = await upsertCompanies(all);
  return NextResponse.json({ discovered: all.length, upserted: inserted });
}

// src/lib/signals/intent/g2-scraper.ts
// Scrapes G2/Capterra for negative reviews of competitor products.
// A company leaving a 2-3 star competitor review is actively dissatisfied = hot signal.

import { runActor, getRunStatus, getDatasetItems } from '@/lib/apify';
import type { ParsedSignal } from '@/lib/gtm/types';

const COMPETITOR_G2_URLS = (process.env.COMPETITOR_G2_URLS || '').split(',').filter(Boolean);

export async function scrapeCompetitorReviews(): Promise<ParsedSignal[]> {
  if (!COMPETITOR_G2_URLS.length) return [];

  const { runId, datasetId } = await runActor('apify/cheerio-scraper', {
    startUrls: COMPETITOR_G2_URLS.map((url) => ({ url: url.trim() })),
    pageFunction: `async function pageFunction({ $, request }) {
      const reviews = [];
      $('.review-card, [itemprop="review"], .paper.paper--white.paper--shadow').each((_, el) => {
        const stars = parseInt($(el).find('[data-rating], .rating, .stars').attr('data-rating') || '5');
        const reviewerTitle = $(el).find('.reviewer-title, [data-test="title"]').text().trim();
        const company = $(el).find('.reviewer-company, [data-test="company"]').text().trim();
        const text = $(el).find('.review-text, [data-test="review-body"]').text().trim().slice(0, 300);
        if (stars <= 3 && company) reviews.push({ stars, reviewerTitle, company, text, sourceUrl: request.url });
      });
      return reviews;
    }`,

    .map((e) => e.linkedin_url)

// src/lib/heat/account-scorer.ts
// Aggregates all signals for each company into a heat score.
// Runs daily. Score = sum of signal scores (with recency decay) capped at 200.

import { supabase } from '@/lib/supabase/client';

export async function recalculateHeatScores(): Promise<number> {
  // Get all companies with signals in the last 60 days
  const { data: companies } = await supabase
    .from('companies')
    .select('id, domain')
    .limit(500);

  if (!companies?.length) return 0;

  let updated = 0;
  const now = Date.now();

  for (const company of companies) {
    const { data: signals } = await supabase
      .from('raw_signals')
      .select('signal_type, signal_date, score, created_at')
      .eq('company_domain', company.domain)
      .gte('created_at', new Date(now - 60 * 24 * 60 * 60 * 1000).toISOString());

    if (!signals?.length) continue;

    const last7d = signals.filter((s) =>
      new Date(s.created_at).getTime() > now - 7 * 24 * 60 * 60 * 1000
    );
    const last30d = signals.filter((s) =>
      new Date(s.created_at).getTime() > now - 30 * 24 * 60 * 60 * 1000
    );

    // Heat score: sum of signal scores with recency boost for last 7 days
    const score = Math.min(
      signals.reduce((sum, s) => {
        const ageDays = (now - new Date(s.created_at).getTime()) / (1000 * 60 * 60 * 24);
        const decay = ageDays <= 7 ? 1 : ageDays <= 30 ? 0.7 : 0.4;
        return sum + (s.score || 50) * decay;
      }, 0),
      300
    );

    const latestSignal = signals.sort((a, b) =>
      new Date(b.created_at).getTime() - new Date(a.created_at).getTime()
    )[0];

    await supabase.from('account_heat_scores').upsert({
      company_id: company.id,
      score: Math.round(score),
      tier: score >= 150 ? 1 : score >= 80 ? 2 : 3,
      signals_7d: last7d.length,
      signals_30d: last30d.length,
      last_signal_type: latestSignal.signal_type,
      last_signal_at: latestSignal.created_at,
      last_calculated_at: new Date().toISOString(),
      updated_at: new Date().toISOString(),
    }, { onConflict: 'company_id' });

    updated++;
  }
  return updated;
}

    if ((count || 0) > 0) continue;

    // Phone number lookup — use a research/find approach
    // For now: log intent. In production, integrate a phone lookup API here.
    const phoneNumber = lead.research_data?.phone_number as string | undefined;
    if (!phoneNumber) {
      // Log voicemail-intent touchpoint for dashboard visibility
      await supabase.from('touchpoints').insert({
        lead_id: lead.id,
        channel: 'voice',
        touch_type: 'voice_call',
        status: 'failed',
        content: 'No phone number available',
      });
      continue;
    }

    try {
      const result = await initiateCall({
        phoneNumber,
        firstName: lead.first_name || '',
        companyName: lead.company_name || '',
        personalizedContext: lead.personalized_opener || lead.signal_summary || '',
      });

      await supabase.from('voice_calls').insert({
        lead_id: lead.id,
        vapi_call_id: result.callId,
        phone_number: phoneNumber,
        status: 'initiated',
      });

      await supabase.from('touchpoints').insert({
        lead_id: lead.id,
        channel: 'voice',
        touch_type: 'voice_call',
        status: 'sent',
        external_id: result.callId,
      });

      called++;
    } catch (err) {
      console.error(`[voice] Call failed for lead ${lead.id}:`, err);
    }
  }

  return called;
}

export async function POST(req: NextRequest) {

  else if (t.includes('call me back') || t.includes('try again')) outcome = 'callback_requested';
  else if (durationSeconds < 5) outcome = 'no_answer';
  else if (durationSeconds < 20) outcome = 'voicemail';

  await supabase.from('voice_calls').update({
    status: 'completed',
    duration_s: durationSeconds,
    transcript,
    outcome,
    ended_at: new Date().toISOString(),
  }).eq('vapi_call_id', callId);

  return NextResponse.json({ ok: true });

    not_now: `Draft a brief, gracious reply (2-3 sentences). Acknowledge their timing, offer to follow up in 30 days, and leave the door open. No pressure.`,

    question: `Draft a helpful, specific answer to their question (3-4 sentences). Be direct and informative. End with a soft call-to-action for a quick call.`,
    other: `Draft a short, professional 2-3 sentence reply acknowledging their message and keeping the conversation open.`,
  };

  const prompt = prompts[params.classification] || prompts.other;

  const msg = await client.messages.create({
    model: 'claude-sonnet-4-6',

  });

export async function routeInboundReply(params: {
  leadId: string | null;
  channel: 'email' | 'linkedin' | 'voice';
  replyText: string;
  fromEmail?: string;
}): Promise<{ classification: string; draftedReply: string | null }> {
  // Get lead context
  let lead = null;
  if (params.leadId) {
    const { data } = await supabase
      .from('pipeline_leads').select('*').eq('id', params.leadId).maybeSingle();
    lead = data;
  } else if (params.fromEmail) {
    const { data } = await supabase
      .from('pipeline_leads').select('*').eq('email', params.fromEmail).maybeSingle();
    lead = data;
  }

  const classification = await classifyReply(params.replyText);

  const drafted = lead ? await draftReply({
    classification,
    inboundText: params.replyText,
    firstName: lead.first_name || 'there',
    companyName: lead.company_name || '',
    signalSummary: lead.signal_summary,
  }) : null;

  // Store in conversations table
  await supabase.from('conversations').insert({
    lead_id: params.leadId || lead?.id || null,
    channel: params.channel,
    inbound_text: params.replyText,
    classification,
    drafted_reply: drafted,
  });

  // If unsubscribe: mark lead as filtered
  if (classification === 'unsubscribe' && lead) {
    await supabase.from('pipeline_leads')
      .update({ status: 'filtered', failure_reason: 'unsubscribed' })
      .eq('id', lead.id);
  }

  return { classification, draftedReply: drafted };
}

// src/lib/orchestrator/sequence-runner.ts
// After email push: trigger LinkedIn connect immediately.
// After 5 days with no reply: trigger LinkedIn DM + schedule voice call.
// This runs as part of the daily cron.

import { supabase } from '@/lib/supabase/client';
import { sendLinkedInConnect, sendFollowUpDM } from '@/lib/linkedin/connection-sender';
import { scheduleFollowUpCalls } from '@/lib/voice/call-scheduler';

export async function runDailySequenceJobs(): Promise<{
  linkedinConnects: number;
  linkedinDMs: number;
  voiceCalls: number;
}> {
  let linkedinConnects = 0;
  let linkedinDMs = 0;

  // 1. Send LinkedIn connection requests for leads pushed in the last 24h
  const yesterday = new Date(Date.now() - 24 * 60 * 60 * 1000).toISOString();
  const { data: newLeads } = await supabase
    .from('pipeline_leads')
    .select('*')
    .eq('status', 'pushed')
    .gte('pushed_at', yesterday)
    .not('linkedin_url', 'is', null)
    .limit(20);

  for (const lead of newLeads || []) {
    // Check no LinkedIn touchpoint already
    const { count } = await supabase
      .from('touchpoints')
      .select('*', { count: 'exact', head: true })
      .eq('lead_id', lead.id)
      .eq('channel', 'linkedin');

    if ((count || 0) === 0) {
      const success = await sendLinkedInConnect(lead).catch(() => false);
      if (success) linkedinConnects++;
    }
  }

  // 2. Send LinkedIn DM follow-up for leads pushed 5 days ago with no email reply
  const fiveDaysAgo = new Date(Date.now() - 5 * 24 * 60 * 60 * 1000).toISOString();
  const fourDaysAgo = new Date(Date.now() - 4 * 24 * 60 * 60 * 1000).toISOString();

  const { data: staleLeads } = await supabase
    .from('pipeline_leads')
    .select('*')
    .eq('status', 'pushed')
    .lt('pushed_at', fiveDaysAgo)
    .gte('pushed_at', fourDaysAgo) // window: day 5 only
    .not('linkedin_url', 'is', null)
    .limit(20);

  for (const lead of staleLeads || []) {
    const { count: dmCount } = await supabase
      .from('touchpoints')
      .select('*', { count: 'exact', head: true })
      .eq('lead_id', lead.id)
      .eq('touch_type', 'linkedin_dm');

    if ((dmCount || 0) === 0) {
      const success = await sendFollowUpDM(lead).catch(() => false);
      if (success) linkedinDMs++;
    }
  }

  // 3. Voice call follow-ups (day 5+ leads)
  const voiceCalls = await scheduleFollowUpCalls().catch(() => 0);

  return { linkedinConnects, linkedinDMs, voiceCalls };
}
